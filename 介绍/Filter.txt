package com.filter;

import javax.servlet.Filter;
import javax.servlet.FilterChain;
import javax.servlet.FilterConfig;
import javax.servlet.ServletException;
import javax.servlet.ServletRequest;
import javax.servlet.ServletResponse;
import java.io.IOException;

/**
 * 基础过滤器（XML配置版）
 * 实现 Filter 接口，重写三大生命周期方法
 */
public class MyBasicFilter implements Filter {

    // 初始化方法：服务器启动时执行，仅1次
    @Override
    public void init(FilterConfig filterConfig) throws ServletException {
        System.out.println("【过滤器】init 初始化完成");
    }

    // 核心拦截方法：每次请求被拦截都会执行
    @Override
    public void doFilter(ServletRequest request,
                         ServletResponse response,
                         FilterChain chain) throws IOException, ServletException {

        // 请求到达目标资源 之前 执行的代码
        System.out.println("【过滤器】拦截到请求，执行前置处理");

        // 核心：放行请求，必须编写！否则请求终止
        chain.doFilter(request, response);

        // 目标资源执行完毕、响应返回浏览器 之前 执行的代码
        System.out.println("【过滤器】目标资源执行完毕，执行后置处理");
    }

    // 销毁方法：服务器正常关闭时执行，仅1次
    @Override
    public void destroy() {
        System.out.println("【过滤器】destroy 过滤器销毁");
    }
}

package com.filter;

import javax.servlet.Filter;
import javax.servlet.FilterChain;
import javax.servlet.FilterConfig;
import javax.servlet.ServletException;
import javax.servlet.ServletRequest;
import javax.servlet.ServletResponse;
import java.io.IOException;

public class MyBasicFilter implements Filter {
    // 定义变量接收web.xml中配置的初始化参数
    private String encoding;
    private String filterDesc;

    @Override
    public void init(FilterConfig filterConfig) throws ServletException {
        // 从FilterConfig对象中读取web.xml配置的初始化参数
        encoding = filterConfig.getInitParameter("encoding");
        filterDesc = filterConfig.getInitParameter("filterDesc");

        // 打印参数，验证读取成功
        System.out.println("【过滤器初始化】读取到配置参数：");
        System.out.println("编码参数：" + encoding);
        System.out.println("过滤器描述：" + filterDesc);
    }

    @Override
    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws IOException, ServletException {
        // 示例：使用读取到的编码参数设置请求/响应编码
        if (encoding != null) {
            request.setCharacterEncoding(encoding);
            response.setContentType("text/html;charset=" + encoding);
        }

        System.out.println("【过滤器】拦截到请求，执行前置处理");
        chain.doFilter(request, response);
        System.out.println("【过滤器】目标资源执行完毕，执行后置处理");
    }

    @Override
    public void destroy() {
        System.out.println("【过滤器】销毁");
    }
}
package com.filter;

import javax.servlet.*;
import java.io.IOException;

/**
 * 统一字符编码过滤器：解决项目全局中文乱码问题
 * XML配置版，支持从web.xml读取编码参数，避免硬编码
 */
public class EncodingFilter implements Filter {
    // 接收web.xml中配置的编码参数
    private String encoding;

    @Override
    public void init(FilterConfig filterConfig) throws ServletException {
        // 读取配置文件中的编码参数，默认UTF-8
        encoding = filterConfig.getInitParameter("encoding");
        if (encoding == null || encoding.trim().isEmpty()) {
            encoding = "UTF-8";
        }
        System.out.println("【编码过滤器】初始化完成，使用编码：" + encoding);
    }

    @Override
    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws IOException, ServletException {
        // ========== 核心：统一设置请求/响应编码 ==========
        // 设置POST请求编码（解决请求参数中文乱码）
        request.setCharacterEncoding(encoding);
        // 设置响应编码与内容类型（解决页面输出中文乱码）
        response.setContentType("text/html;charset=" + encoding);

        // 放行请求，让请求继续访问目标资源
        chain.doFilter(request, response);
    }

    @Override
    public void destroy() {
        System.out.println("【编码过滤器】销毁");
    }
}
